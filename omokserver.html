OmokServer

ASP.NET Core Web API와 SignalR을 활용한 온라인 오목 게임 서버입니다. RESTful API를 통한 사용자 관리와 SignalR을 통한 실시간 게임플레이를 구현했습니다.

C# ASP.NET Core SignalR JWT MySQL BCrypt
← 프로젝트 목록으로 돌아가기

프로젝트 개요

OmokServer는 현대적인 웹 기술을 활용하여 구현한 완전한 온라인 오목 게임 서버입니다. 사용자 인증, 매치메이킹, 실시간 게임플레이, 경기 결과 저장 등 온라인 게임에 필요한 모든 기능을 포함합니다.

프로젝트 시연 영상

실제 게임 플레이와 주요 기능들을 영상으로 확인해보세요!

핵심 기능

🔐

JWT 인증 시스템

BCrypt를 이용한 안전한 비밀번호 해싱과 JWT Bearer 토큰 기반의 사용자 인증 시스템을 구현했습니다.

⚡

SignalR 실시간 통신

SignalR Hub를 통한 양방향 실시간 통신으로 부드러운 멀티플레이어 게임 경험을 제공합니다.

🎯

공정한 매치메이킹

ConcurrentQueue와 락을 사용한 안전하고 공정한 매치메이킹 시스템으로 플레이어들을 자동으로 매칭합니다.

🎮

완전한 게임 로직

오목 게임의 모든 규칙을 구현했습니다. 승패 판정, 턴 관리, 게임 상태 검증 등을 자동으로 처리합니다.

📊

경기 결과 관리

모든 경기 결과를 데이터베이스에 저장하고, 사용자별 전적 조회 기능을 제공합니다.

🏗️

계층형 아키텍처

DTO, Repository, Service 패턴을 적용한 깔끔한 코드 구조로 유지보수성과 확장성을 확보했습니다.

사용자 인증

회원가입 또는 로그인을 통해 JWT 토큰을 발급받습니다.

승패 판정 & 결과 저장

GameHub 구현

[Authorize]
public class GameHub : Hub
{
    private readonly IMatchmakingService _matchmakingService;
    private readonly GameRoomManager _gameRoomManager;
    
    // 사용자 등록
    public async Task RegisterUser()
    {
        var userId = Context.UserIdentifier;
        var connectionId = Context.ConnectionId;
        
        await _userConnectionManager.AddConnection(userId, connectionId);
        await Groups.AddToGroupAsync(connectionId, $"User_{userId}");
    }
    
    // 오목돌 놓기
    public async Task PlaceStone(int gameId, int x, int y)
    {
        var userId = Context.UserIdentifier;
        var gameRoom = _gameRoomManager.GetGameRoom(gameId);
        
        if (gameRoom?.IsPlayerTurn(userId) == true)
        {
            var result = gameRoom.PlaceStone(userId, x, y);
            
            // 모든 플레이어에게 수 정보 전송
            await Clients.Group($"Game_{gameId}")
                .InvokeAsync("StonePlaced", x, y, userId);
            
            // 게임 종료 확인
            if (result.IsGameOver)
            {
                await HandleGameOver(gameRoom, result);
            }
        }
    }
    
    // 게임 종료 처리
    private async Task HandleGameOver(GameRoom gameRoom, GameResult result)
    {
        // 데이터베이스에 결과 저장
        await _matchHistoryService.SaveMatch(new Match
        {
            Player1Id = gameRoom.Player1Id,
            Player2Id = gameRoom.Player2Id,
            WinnerId = result.WinnerId,
            StartTime = gameRoom.StartTime,
            EndTime = DateTime.UtcNow,
            Moves = string.Join(";", gameRoom.Moves)
        });
        
        // 클라이언트에 게임 종료 알림
        await Clients.Group($"Game_{gameRoom.Id}")
            .InvokeAsync("GameOver", result);
    }
}

핵심 구현 코드

매치메이킹 서비스

public class MatchmakingService : IMatchmakingService
{
    private readonly ConcurrentQueue<string> _waitingPlayers = new();
    private readonly ConcurrentDictionary<string, DateTime> _queueTimes = new();
    private readonly object _matchmakingLock = new();
    
    public async Task<bool> JoinQueue(string userId)
    {
        lock (_matchmakingLock)
        {
            if (_queueTimes.ContainsKey(userId))
                return false; // Already in queue
                
            _waitingPlayers.Enqueue(userId);
            _queueTimes[userId] = DateTime.UtcNow;
            
            // Try to make a match
            return TryCreateMatch();
        }
    }
    
    private bool TryCreateMatch()
    {
        if (_waitingPlayers.Count >= 2)
        {
            if (_waitingPlayers.TryDequeue(out string player1) &&
                _waitingPlayers.TryDequeue(out string player2))
            {
                // Create new game room
                var gameRoom = _gameRoomManager.CreateGameRoom(player1, player2);
                
                // Notify players
                _ = Task.Run(async () =>
                {
                    await _hubContext.Clients.User(player1)
                        .InvokeAsync("MatchFound", gameRoom.Id, player2);
                    await _hubContext.Clients.User(player2)
                        .InvokeAsync("MatchFound", gameRoom.Id, player1);
                });
                
                // Remove from queue tracking
                _queueTimes.TryRemove(player1, out _);
                _queueTimes.TryRemove(player2, out _);
                
                return true;
            }
        }
        return false;
    }
}

게임룸 관리

public class GameRoomManager
{
    private readonly ConcurrentDictionary<int, GameRoom> _gameRooms = new();
    private int _nextGameId = 1;
    
    public GameRoom CreateGameRoom(string player1Id, string player2Id)
    {
        var gameId = Interlocked.Increment(ref _nextGameId);
        var gameRoom = new GameRoom
        {
            Id = gameId,
            Player1Id = player1Id,
            Player2Id = player2Id,
            Board = new int[15, 15], // 15x15 오목판
            CurrentTurn = player1Id,
            StartTime = DateTime.UtcNow,
            Status = GameStatus.InProgress
        };
        
        _gameRooms[gameId] = gameRoom;
        return gameRoom;
    }
    
    public GameResult PlaceStone(int gameId, string playerId, int x, int y)
    {
        if (!_gameRooms.TryGetValue(gameId, out var gameRoom))
            return GameResult.InvalidGame();
            
        // 턴 확인
        if (gameRoom.CurrentTurn != playerId)
            return GameResult.NotYourTurn();
            
        // 보드 범위 확인
        if (x < 0 || x >= 15 || y < 0 || y >= 15)
            return GameResult.InvalidPosition();
            
        // 이미 돌이 놓여있는지 확인
        if (gameRoom.Board[x, y] != 0)
            return GameResult.PositionOccupied();
            
        // 돌 놓기
        int stoneColor = gameRoom.Player1Id == playerId ? 1 : 2;
        gameRoom.Board[x, y] = stoneColor;
        gameRoom.Moves.Add($"{x},{y}");
        
        // 승리 조건 확인
        if (CheckWinCondition(gameRoom.Board, x, y, stoneColor))
        {
            gameRoom.Status = GameStatus.Finished;
            gameRoom.WinnerId = playerId;
            gameRoom.EndTime = DateTime.UtcNow;
            
            return GameResult.GameWon(playerId);
        }
        
        // 턴 변경
        gameRoom.CurrentTurn = gameRoom.Player1Id == playerId 
            ? gameRoom.Player2Id : gameRoom.Player1Id;
            
        return GameResult.Success();
    }
    
    private bool CheckWinCondition(int[,] board, int x, int y, int color)
    {
        // 4방향(가로, 세로, 대각선 2개)으로 5목 확인
        int[,] directions = { {1,0}, {0,1}, {1,1}, {1,-1} };
        
        for (int dir = 0; dir < 4; dir++)
        {
            int count = 1; // 현재 돌 포함
            int dx = directions[dir, 0];
            int dy = directions[dir, 1];
            
            // 양방향으로 같은 색 돌 개수 세기
            count += CountStonesInDirection(board, x, y, dx, dy, color);
            count += CountStonesInDirection(board, x, y, -dx, -dy, color);
            
            if (count >= 5) return true;
        }
        
        return false;
    }
}
